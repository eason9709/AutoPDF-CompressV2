# PDF工具箱運行命令

## 網頁程式運行指令
# 直接使用BAT檔案運行（推薦方式）
run_pdf_toolkit.bat

# 或手動運行以下命令
python -m pip install -r requirements.txt
streamlit run app.py

## 桌面應用打包指令
# 進入應用程序目錄
cd pdf-toolkit-app

# 安裝打包依賴
python -m pip install -r requirements-app.txt

# 創建應用圖標
python create_app_icon.py

# 執行打包命令
python pyinstaller_build.py

# 打包後的文件位於 pdf-toolkit-app/dist/PDF工具箱/ 目錄中
# 運行方式: 雙擊 PDF工具箱.exe 或 PDF工具箱.bat

## 打包後分發方式
